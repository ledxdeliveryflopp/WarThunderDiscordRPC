## Настройки приложения
```yaml
settings:
  api: 
    api_ip: '127.0.0.1' # ip API war tunder
    api_port: 8111 #  порт API war tunder
    indicators_endpoint: '/indicators' # эндпоинт индикаторов техники
    map_endpoint: '/map_info.json' # эндпоинт информации о карте
    air_info_endpoint: '/state'  # эндпоинт информации о воздушной техники
  presence:
    show_indicators: True # True | False показывать ли индикаторы техники
    lang: ru # en | ru язык
    logo_theme: main_red # main_red | main_white вариант логотипа
    custom_images: True # кастомные изображения для некоторой техники
  air_indicators:
    air_speed_type: IAS # IAS | TAS выводить в индикаторы скорость IAS или TAS
    altitude_type: ABSOLUTE  # ABSOLUTE  | RADIO выводить в индикаторы высоту относительно моря или по радиовысотомеру
  logger:
    level: 'INFO' # Уровень логирования
  core:
    timeout: 50 # время ожидания между попытками поставить статус из-за закрытия игры или дискорда
    show_notification: True # Показывать уведомления об доступных обновлениях приложения
```
4. Запустите exe
